-- FUNGSI DAMAGE MAX (PAS BUAT GAME INI)
local Aktif = false
B1.MouseButton1Click:Connect(function()
    Aktif = not Aktif
    B1.BackgroundColor3 = Aktif and Color3.fromRGB(220,60,60) or Color3.fromRGB(150,40,40)
    
    if Aktif then
        -- UBAH KEKUATAN JADI TAK TERBATAS
        spawn(function()
            while Aktif do wait()
                pcall(function()
                    -- Ambil statistik & ganti jadi angka paling besar
                    local leaderstats = Player:FindFirstChild("leaderstats") or Player:FindFirstChild("Stats")
                    if leaderstats then
                        if leaderstats:FindFirstChild("Strength") then leaderstats.Strength.Value = 9e99 end
                        if leaderstats:FindFirstChild("Kekuatan") then leaderstats.Kekuatan.Value = 9e99 end
                        if leaderstats:FindFirstChild("Power") then leaderstats.Power.Value = 9e99 end
                    end
                    
                    -- Tambah Damage Lewat Remote
                    for _,v in pairs(ReplicatedStorage:GetDescendants()) do
                        if v:IsA("RemoteEvent") and (v.Name:lower():find("hit") or v.Name:lower():find("attack") or v.Name:lower():find("damage")) then
                            v.OnServerEvent:Connect(function(plr,tar,dmg)
                                if plr == Player then
                                    v:FireServer(tar, 999999999) -- PAKAI ANGKA PALING BESAR
                                end
                            end)
                        end
                    end
                end)
            end
        end)
    end
end)

-- FUNGSI KEBAL
B2.MouseButton1Click:Connect(function()
    local Char = Player.Character or Player.CharacterAdded:Wait()
    local Hum = Char:WaitForChild("Humanoid")
    Hum.MaxHealth = math.huge
    Hum.Health = math.huge
    Hum.HealthChanged:Connect(function() Hum.Health = Hum.MaxHealth end)
    B2.Text = "✅ SUDAH KEBAL"
end)

-- FUNGSI KENCANG

    Hum.JumpPower = 100
    B3.Text = "✅ SUDAH KENCANG"
end)
