-- // FUNGSI BIKIN TOMBOL //
local function BuatTombol(Nama, PosY)
    local Btn = Instance.new("Frame")
    Btn.Parent = MainFrame
    Btn.BackgroundColor3 = Color3.fromRGB(25,25,25)
    Btn.Position = UDim2.new(0.1, 0, PosY, 0)
    Btn.Size = UDim2.new(0.8, 0, 0, 50)

    local GarisAktif = Instance.new("Frame")
    GarisAktif.Parent = Btn
    GarisAktif.BackgroundColor3 = Color3.fromRGB(80, 120, 255)
    GarisAktif.Position = UDim2.new(0, 0, 0.2, 0)
    GarisAktif.Size = UDim2.new(0, 4, 0.6, 0)
    GarisAktif.Visible = false

    local Txt = Instance.new("TextLabel")
    Txt.Parent = Btn
    Txt.BackgroundTransparency = 1
    Txt.Position = UDim2.new(0.1, 0, 0, 0)
    Txt.Size = UDim2.new(0.8, 0, 1, 0)
    Txt.Font = Enum.Font.GothamSemibold
    Txt.Text = Nama
    Txt.TextColor3 = Color3.fromRGB(180,180,180)
    Txt.TextSize = 14

    -- Status Aktif/Mati
    local Aktif = false
    Btn.InputBegan:Connect(function(Input)
        if Input:Is("MouseButton1") then
            Aktif = not Aktif
            GarisAktif.Visible = Aktif
            Txt.TextColor3 = Aktif and Color3.new(1,1,1) or Color3.fromRGB(180,180,180)
            Btn.BackgroundColor3 = Aktif and Color3.fromRGB(30,30,30) or Color3.fromRGB(25,25,25)
        end
    end)

    return {Btn=Btn, Aktif=function() return Aktif end}
end

-- // BIKIN TOMBOL FITUR //
local TombolAutoX2 = BuatTombol("🤖 AUTO KLIK x2", 0.2)
local TombolKebal = BuatTombol("🛡️ KEBAL / GOD MODE", 0.48)

-- // FITUR 1: AUTO KLIK x2 //
spawn(function()
    while wait(0.05) do
        if TombolAutoX2.Aktif() then
            pcall(function()
                local Karakter = Player.Character
                if not Karakter then return end

                -- Klik Tombol Tangan / x2 di Layar
                for _, Gui in pairs(Player.PlayerGui:GetDescendants()) do
                    if Gui:IsA("ImageButton") or Gui:IsA("TextButton") then
                        if Gui.Visible and (Gui.Name:lower():find("multi") or Gui.Name:lower():find("x2") or Gui.Name:lower():find("bonus")) then
                            fireclickdetector(Gui)
                        end
                    end
                end

                -- Klik Tombol di Dunia
                for _, Benda in pairs(Workspace:GetDescendants()) do
                    if Benda:IsA("Part") and Benda:FindFirstChild("ClickDetector") then
                        if Benda.Name:lower():find("x2") or Benda.Name:lower():find("multiplier") or Benda.Name:lower():find("pop") then
                            fireclickdetector(Benda.ClickDetector)
                        end
                    end
                end
            end)
        end
    end
end)

-- // FITUR 2: KEBAL / GOD MODE //
spawn(function()
    while wait() do
        if TombolKebal.Aktif() then
            pcall(function()
                local Karakter = Player.Character or Player.CharacterAdded:Wait()
                local Humanoid = Karakter:WaitForChild("Humanoid")
                Humanoid.MaxHealth = math.huge
                Humanoid.Health = math.huge
                Humanoid.HealthChanged:Connect(function()
                    if Humanoid.Health < Humanoid.MaxHealth then
                        Humanoid.Health = Humanoid.MaxHealth
                    end
                end)
            end)
        end
    end
end)
