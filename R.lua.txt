local Players = game:GetService("Players")
local UIS = game:GetService("UserInputService") -- buat pencet layar
local Player = Players.LocalPlayer

spawn(function()
    while wait(0.03) do
        pcall(function()
            for _,v in pairs(Player.PlayerGui:GetDescendants()) do
                if (v:IsA("ImageButton") or v:IsA("TextButton")) and v.Visible then
                    -- ✅ SYARAT LENGKAP & RAPI
                    if string.find(string.lower(v.Name), "x2")
                    or string.find(string.lower(tostring(v.Image)), "hand")
                    or v.ImageColor3 == Color3.fromRGB(170, 0, 255) then

                        fireclickdetector(v) -- cara 1
                        v.MouseButton1Down:Fire() -- cara 2
                        wait(0.01)
                        v.MouseButton1Up:Fire()
                        UIS:Tap(v.AbsolutePosition + v.AbsoluteSize/2) -- cara 3: klik tengah

                        print("✅ BERHASIL PENCET X2!")
                    end
                end
            end
        end)
    end
end)

print("✅ SCRIPT SIAP JALAN!")
