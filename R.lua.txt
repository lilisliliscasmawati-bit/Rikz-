-- // FUNGSI BIKIN TOMBOL //
local function Tombol(Nama, Y)
    local Btn = Instance.new("TextButton", Main)
    Btn.BackgroundColor3 = Color3.fromRGB(22,22,22)
    Btn.Position = UDim2.new(0.1,0,Y,0)
    Btn.Size = UDim2.new(0.8,0,0,45)
    Btn.Font = Enum.Font.GothamSemibold
    Btn.Text = Nama
    Btn.TextColor3 = Color3.fromRGB(170,170,170)
    Btn.TextSize = 13
    Btn.AutoButtonColor = false

    local GarisKecil = Instance.new("Frame", Btn)
    GarisKecil.BackgroundColor3 = Color3.fromRGB(70, 120, 255)
    GarisKecil.Position = UDim2.new(0,0,0.15,0)
    GarisKecil.Size = UDim2.new(0,3,0.7,0)
    GarisKecil.Visible = false

    local Aktif = false
    Btn.MouseButton1Click:Connect(function()
        Aktif = not Aktif
        GarisKecil.Visible = Aktif
        Btn.TextColor3 = Aktif and Color3.new(1,1,1) or Color3.fromRGB(170,170,170)
        Btn.BackgroundColor3 = Aktif and Color3.fromRGB(30,30,30) or Color3.fromRGB(22,22,22)
    end)

    return {Btn = Btn, Aktif = function() return Aktif end}
end

-- // BIKIN TOMBOL FITUR //
local B1 = Tombol("🤖 AUTO KLIK x2", 0.22)
local B2 = Tombol("🛡️ KEBAL MODE", 0.52)

-- // FITUR 1: AUTO x2 - CARA TERAMPUH (KLIK TENGAH LAYAR LANGSUNG) //
spawn(function()
    while wait(0.01) do -- Klik sangat cepat
        if B1.Aktif() then
            pcall(function()
                -- 🔹 CARA 1: KLIK LANGSUNG DI POSISI TENGAH LAYAR 🔹
                -- Koordinat ini pas banget di tempat tombol ungu kamu muncul
                UIS:InputBegan:Fire({UserInputType = Enum.UserInputType.MouseButton1, Position = Vector2.new(640, 360)})
                UIS:InputEnded:Fire({UserInputType = Enum.UserInputType.MouseButton1, Position = Vector2.new(640, 360)})

                -- 🔹 CARA 2: CARI BERDASARKAN GAMBAR TANGANNYA 🔹
                for _, gui in pairs(Player.PlayerGui:GetDescendants()) do
                    if gui:IsA("ImageButton") and gui.Visible then
                        -- Cari gambar tangan ungu / gambar x2
                        if gui.Image == "rbxassetid://14835872345" or string.find(gui.Image:lower(), "hand") or string.find(gui.Image:lower(), "click") or string.find(gui.Name:lower(), "x2") then
                            fireclickdetector(gui)
                        end
                        -- Cari ukuran besar
                        if gui.AbsoluteSize.X > 100 and gui.AbsoluteSize.Y > 100 then
                            fireclickdetector(gui)
                        end
                    end
                end
            end)
        end
    end
end)

-- // FITUR 2: KEBAL //
spawn(function()
    while wait() do
        if B2.Aktif() then
            pcall(function()
                local Char = Player.Character or Player.CharacterAdded:Wait()
                local Hum = Char:FindFirstChildWhichIsA("Humanoid")
                if Hum and Hum.Health then
                    Hum.MaxHealth = math.huge
                    Hum.Health = math.huge
                    -- Jaga-jaga kalau berubah nama
                    Hum.HealthChanged:Connect(function()
                        Hum.Health = Hum.MaxHealth
                    end)
                end
            end)
        end
    end
end)
