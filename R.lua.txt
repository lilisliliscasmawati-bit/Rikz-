-- Tora Hub | Kick a Lucky Block + AUTO TRAIN
local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/ToraIsMe/ToraIsMe/main/UiLib"))()
local Window = Library:CreateWindow("🎁 KICK A LUCKY BLOCK | TORA HUB", 550, 400)

local TabUtama = Window:CreateTab("🏠 UTAMA")
local TabFitur = Window:CreateTab("⚙️ FITUR")
local TabInfo = Window:CreateTab("ℹ️ INFO")

local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Workspace = game:GetService("Workspace")
local UIS = game:GetService("UserInputService")
local Player = Players.LocalPlayer
local Character, Humanoid, RootPart

-- Update Karakter
local function UpdateChar()
    Character = Player.Character or Player.CharacterAdded:Wait()
    Humanoid = Character:WaitForChild("Humanoid", 10)
    RootPart = Character:WaitForChild("HumanoidRootPart", 10)
end
UpdateChar()
Player.CharacterAdded:Connect(UpdateChar)

-- 🎯 PERFECT KICK
getgenv().PerfectKick = false
TabFitur:CreateToggle({
    Name = "🎯 PERFECT KICK",
    CurrentValue = false,
    Callback = function(state)
        PerfectKick = state
        if state then
            spawn(function()
                while PerfectKick do
                    wait(0.05)
                    pcall(function()
                        local Remote = ReplicatedStorage:WaitForChild("Remotes", 10):WaitForChild("Kick")
                        if Remote and RootPart then Remote:FireServer(1000000) end
                    end)
                end
            end)
        end
    end
})

-- 🚆 AUTO TRAIN (BARU DITAMBAH!)
getgenv().AutoTrain = false
TabFitur:CreateToggle({
    Name = "🚆 AUTO TRAIN / NAIK KEKUATAN",
    CurrentValue = false,
    Callback = function(state)
        AutoTrain = state
        if state then
            spawn(function()
                while AutoTrain do
                    wait(0.02)
                    pcall(function()
                        -- Klik tombol latihan / angkat beban
                        for _,v in pairs(Character:GetDescendants()) do
                            if v:IsA("ProximityPrompt") and (v.Name:find("Train") or v.Name:find("Lift") or v.Name:find("Weight")) then
                                fireproximityprompt(v)
                            end
                        end
                        -- Klik tombol bonus latihan
                        for _,v in pairs(Player.PlayerGui:GetDescendants()) do
                            if v:IsA("ImageButton") and v.Visible and (v.Name:find("Train") or v.Name:find("Lift") or v.Name:find("Bonus")) then
                                fireclickdetector(v)
                            end
                        end
                        -- Klik otomatis kalau ada tombol tekan
                        UIS:InputBegan:Fire({UserInputType=Enum.UserInputType.MouseButton1,Position=Vector2.new(640,360)})
                        UIS:InputEnded:Fire({UserInputType=Enum.UserInputType.MouseButton1,Position=Vector2.new(640,360)})
                    end)
                end
            end)
        end
    end
})
