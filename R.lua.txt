-- ✅ AUTO TRAIN | TANPA MENU | LANGSUNG JALAN
local Players = game:GetService("Players")
local Workspace = game:GetService("Workspace")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Player = Players.LocalPlayer
local Character, RootPart, Remotes

Remotes = ReplicatedStorage:WaitForChild("Remotes")

local function UpdateChar()
    Character = Player.Character or Player.CharacterAdded:Wait()
    RootPart = Character:WaitForChild("HumanoidRootPart")
end
UpdateChar()
Player.CharacterAdded:Connect(UpdateChar)

-- ✅ LANGSUNG LATIHAN TERUS
spawn(function()
    while true do
        wait(0.2)
        pcall(function()
            if not RootPart then return end
            local Tempat = Workspace:FindFirstChild("Training")
            if Tempat then
                for _,v in pairs(Tempat:GetDescendants()) do
                    if v:IsA("Part") and v.Name == "TrainPart" then
                        RootPart.CFrame = v.CFrame+Vector3.new(0,2,0)
                        Remotes.Train:FireServer()
                    end
                end
            end
        end)
    end
end)
