local Rayfield = loadstring(game:HttpGet('https://raw.githubusercontent.com/shahriar0172/Rayfield/main/source'))()

local Window = Rayfield:CreateWindow({
   Name = "Rikz| Lari block keberuntungan",
   Icon = 0,
   LoadingTitle = "Gumanba Hub",
   LoadingSubtitle = "by Gumanba",
   Theme = "Default",
   DisableRayfieldPrompts = true,
   DisableBuildWarnings = true,
   ConfigurationSaving = {
      Enabled = false
   },
   Discord = {
      Enabled = false
   }
})

local Tab = Window:CreateTab("Fitur", 4483362458)
local Section = Tab:CreateSection("Utama")

-- Variabel
local plr = game:GetService("Players").LocalPlayer
local char, hum, root
local Workspace = game:GetService("Workspace")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local remotes = ReplicatedStorage:WaitForChild("Remotes")

-- Update Karakter
local function UpdateChar()
    char = plr.Character or plr.CharacterAdded:Wait()
    hum = char:WaitForChild("Humanoid")
    root = char:WaitForChild("HumanoidRootPart")
end
UpdateChar()
plr.CharacterAdded:Connect(UpdateChar)

-- Perfect Kick
local PerfectKick = false
Tab:CreateToggle({
   Name = "Perfect Kick",
   CurrentValue = false,
   Callback = function(Value)
      PerfectKick = Value
   end
})

-- Auto Train
local AutoTrain = false
Tab:CreateToggle({
   Name = "Auto Train",
   CurrentValue = false,
   Callback = function(Value)
      AutoTrain = Value
   end
})

-- Auto Rebirth
local AutoRebirth = false
Tab:CreateToggle({
   Name = "Auto Rebirth",
   CurrentValue = false,
   Callback = function(Value)
      AutoRebirth = Value
   end
})

-- Auto Collect
local AutoCollect = false
Tab:CreateToggle({
   Name = "Auto Collect Cash",
   CurrentValue = false,
   Callback = function(Value)
      AutoCollect = Value
   end
})

-- Upgrade All
Tab:CreateButton({
   Name = "Upgrade All",
   Callback = function()
      pcall(function()
         remotes.Upgrade:InvokeServer("Power")
         remotes.Upgrade:InvokeServer("Speed")
         remotes.Upgrade:InvokeServer("Strength")
      end)
   end
})

-- LOOP UTAMA
spawn(function()
    while wait(0.1) do
        pcall(function()
            
            -- Perfect Kick
            if PerfectKick and char and root then
                local block = Workspace:FindFirstChild("LuckyBlock")
                if block and block:FindFirstChild("RootPart") then
                    root.CFrame = CFrame.new(block.RootPart.Position + Vector3.new(0, 3, 5))
                    remotes.Kick:FireServer(block.RootPart)
                end
            end

            -- Auto Train
            if AutoTrain then
                local train = Workspace:FindFirstChild("Training")
                if train then
                    for _,v in pairs(train:GetDescendants()) do
                        if v:IsA("Part") and v.Name == "TrainPart" then
                            root.CFrame = v.CFrame + Vector3.new(0,2,0)
                            remotes.Train:FireServer()
                        end
                    end
                end
            end

            -- Auto Rebirth
            if AutoRebirth then
                if plr.leaderstats.Rebirths.Value < 1000 then
                    remotes.Rebirth:InvokeServer()
                end
            end

            -- Auto Collect
            if AutoCollect then
                for _,v in pairs(Workspace:GetDescendants()) do
                    if v:IsA("Part") and v.Name == "Cash" then
                        root.CFrame = v.CFrame + Vector3.new(0,2,0)
                        remotes.Collect:FireServer(v)
                    end
                end
            end

        end)
    end
end)

Rayfield:Notify({
   Title = "Berhasil!",
   Content = "Script Gumanba Dimuat",
   Duration = 3
})
