-- AUTO x2 KE MANAPUN PINDAH + KEBAL --

-- 🔹 MUAT TAMPILAN MENU TORA ASLI 🔹
local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/ToraIsMe/ToraIsMe/main/UiLib"))()
local Window = Library:CreateWindow("⚡ RIKZ HUB ⚡", 520, 400)

-- 🔹 BAGIAN MENU 🔹
local MainTab = Window:CreateTab("🏠 UTAMA")
local FarmTab = Window:CreateTab("🤖 OTOMATIS")
local LainTab = Window:CreateTab("🛡️ LAINNYA")

-- 🔹 VARIABEL PENGATURAN 🔹
getgenv().AutoX2 = false
getgenv().Kebal = false

-- ============== FITUR 1: AUTO KLIK x2 ==============
FarmTab:CreateToggle("🤖 AUTO KLIK x2 (KE MANAPUN PINDAH)", function(state)
    AutoX2 = state
    if state then
        spawn(function()
            while AutoX2 do wait(0.01)
                pcall(function()
                    -- 🎯 CARA AMPUH: KLIK TENGAH LAYAR + CARI GAMBAR
                    local UIS = game:GetService("UserInputService")
                    -- Klik titik tengah pas tempat tombol ungu
                    UIS:InputBegan:Fire({UserInputType = Enum.UserInputType.MouseButton1, Position = Vector2.new(640, 360)})
                    UIS:InputEnded:Fire({UserInputType = Enum.UserInputType.MouseButton1, Position = Vector2.new(640, 360)})

                    -- 🔍 CARI BERDASARKAN GAMBAR & UKURAN
                    for _, gui in pairs(game.Players.LocalPlayer.PlayerGui:GetDescendants()) do
                        if gui:IsA("ImageButton") and gui.Visible then
                            if gui.AbsoluteSize.X > 80 and gui.AbsoluteSize.Y > 80 then
                                if string.find(gui.Name:lower(), "x2") or string.find(gui.Image:lower(), "hand") or string.find(gui.Image:lower(), "multi") then
                                    fireclickdetector(gui)
                                end
                            end
                        end
                    end
                end)
            end
        end)
    end
end)

-- ============== FITUR 2: KEBAL / GOD MODE ==============
LainTab:CreateToggle("🛡️ KEBAL MODE (DARAH TAK TERHENTI)", function(state)
    Kebal = state
    if state then
        spawn(function()
            while Kebal do wait()
                pcall(function()
                    local Char = game.Players.LocalPlayer.Character or game.Players.LocalPlayer.CharacterAdded:Wait()
                    local Hum = Char:WaitForChild("Humanoid")
                    Hum.MaxHealth = math.huge
                    Hum.Health = math.huge
                    Hum.HealthChanged:Connect(function() Hum.Health = math.huge end)
                end)
            end
        end)
    end
end)

-- ============== INFORMASI ==============
MainTab:CreateLabel("✅ Script Berhasil Dimuat!")
MainTab:CreateLabel("💡 Fitur sudah disesuaikan dengan game kamu")

-- 🔔 NOTIFIKASI
Library:Notify("🔥 RIKZ HUB SIAP PAKAI!", "TORA STYLE", 3)
