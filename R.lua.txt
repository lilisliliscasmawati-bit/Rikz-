-- ✅ AUTO PENCET X2 SAJA | DIEM DI TEMPAT
local Players = game:GetService("Players")
local Player = Players.LocalPlayer

spawn(function()
    while true do
        wait(0.08)
        pcall(function()
            for _,v in pairs(Player.PlayerGui:GetDescendants()) do
                if (v:IsA("ImageButton") or v:IsA("TextButton")) and v.Visible then
                    if string.find(string.lower(v.Name), "x2") or string.find(string.lower(v.Name), "2x") or string.find(string.lower(v.Name), "double") or string.find(string.lower(v.Name), "multiplier") then
                        fireclickdetector(v)
                        v.MouseButton1Down:Fire()
                        wait(0.02)
                        v.MouseButton1Up:Fire()
                    end
                end
            end
        end)
    end
end)

print("✅ SUDAH AKTIF: PENCET X2 OTOMATIS, DIEM AJA!")
