Tab_Main:AddToggle({
    Name = "🛡️ Mode Kebal",
    Default = false,
    Callback = function(nyala)
        _G.Kebal = nyala
        if nyala then
            OrionLib:MakeNotification({Title="INFO", Content="Mode Kebal AKTIF!", Time=3})
        else
            OrionLib:MakeNotification({Title="INFO", Content="Mode Kebal MATI!", Time=3})
        end
        
        -- LOOP KEAMANAN
        spawn(function()
            while task.wait(0.1) do
                if _G.Kebal then
                    pcall(function()
                        local plr = game:GetService("Players").LocalPlayer
                        local char = plr.Character
                        if char and char:FindFirstChild("Humanoid") then
                            char.Humanoid.Health = math.huge
                            char.Humanoid.MaxHealth = math.huge
                        end
                    end)
                end
            end
        end)
    end
})
