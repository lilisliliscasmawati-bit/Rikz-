local OrionLib = loadstring(game:HttpGet('https://raw.githubusercontent.com/jensonhirst/Orion/main/source'))()

-- BAGIAN WINDOW (DIUBAH SESUAI MAU KAMU)
local Window = OrionLib:MakeWindow({
    Name = "RIKZ HUB", -- Nama menu
    HidePremium = false,
    SaveConfig = true,
    ConfigFolder = "Rikz_Config", -- Nama folder simpan pengaturan
    IntroEnabled = true, -- Tampilkan animasi awal
    IntroText = "Lucky Block Rush", -- Tulisan saat animasi
    IntroIcon = "rbxassetid://4483333458", -- Gambar animasi
    Icon = "rbxassetid://4483333458", -- Gambar di judul menu
    CloseCallback = function() end -- Kerjaan kalau menu ditutup
})
-- BIKIN TAB / HALAMAN
local Tab_Utama = Window:MakeTab({
    Name = "Fitur Utama",
    Icon = "rbxassetid://4483333458"
})
-- CONTOH TOMBOL KEBAL (LENGKAP & BERFUNGSI)
Tab_Utama:AddToggle({
    Name = "🛡️ Mode Kebal",
    Default = false,
    Callback = function(nyala)
        _G.KebalAktif = nyala
        if nyala then
            OrionLib:MakeNotification({Name="AKTIF", Content="Kamu Kebal / Gak Bisa Mati!", Time=2})
            spawn(function()
                while _G.KebalAktif and task.wait(0.1) do
                    pcall(function()
                        local Plr = game.Players.LocalPlayer
                        local Char = Plr.Character or Plr.CharacterAdded:Wait()
                        local Hum = Char:FindFirstChildWhichIsA("Humanoid")
                        if Hum then
                            Hum.Health = math.huge
                            Hum.MaxHealth = math.huge
                        end
                    end)
                end
            end)
        else
            _G.KebalAktif = false
            OrionLib:MakeNotification({Name="MATI", Content="Kebal Dimatikan", Time=2})
        end
    end
})
-- WAJIB DI BAWAH BIAR MENU MUNCUL
OrionLib:Init()
