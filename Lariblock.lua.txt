-- ✨ FUNGSI ATTACK SPEED PALING KENCANG ✨
local SpeedSerang = false
B1.MouseButton1Click:Connect(function()
    SpeedSerang = not SpeedSerang
    B1.BackgroundColor3 = SpeedSerang and Color3.fromRGB(240,80,160) or Color3.fromRGB(180,60,120)
    
    if SpeedSerang then
        spawn(function()
            while SpeedSerang do wait(0.01) -- ⚡ SERANG TIAP 0,01 DETIK!
                pcall(function()
                    -- Otomatis tekan tombol serang terus-menerus
                    UIS:InputBegan:Fire({UserInputType = Enum.UserInputType.MouseButton1})
                    
                    -- Ubah cooldown senjata/jadi 0
                    local Char = Player.Character
                    if Char then
                        for _,v in pairs(Char:GetDescendants()) do
                            if v:IsA("NumberValue") and (v.Name:lower():find("cooldown") or v.Name:lower():find("delay")) then
                                v.Value = 0
                            end
                        end
                    end

                    -- Ubah statistik kecepatan serang
                    local leaderstats = Player:FindFirstChild("leaderstats") or Player:FindFirstChild("Stats")
                    if leaderstats then
                        if leaderstats:FindFirstChild("AttackSpeed") then leaderstats.AttackSpeed.Value = 99999 end
                        if leaderstats:FindFirstChild("Kecepatan") then leaderstats.Kecepatan.Value = 99999 end
                    end
                end)
            end
        end)
    end
end)

-- 🛡️ FUNGSI KEBAL
B2.MouseButton1Click:Connect(function()
    local Char = Player.Character or Player.CharacterAdded:Wait()
    local Hum = Char:WaitForChild("Humanoid")
    Hum.MaxHealth = math.huge
    Hum.Health = math.huge
    Hum.HealthChanged:Connect(function() Hum.Health = Hum.MaxHealth end)
    B2.Text = "✅ SUDAH KEBAL"
end)

-- 🏃 FUNGSI LARI KENCANG
B3.MouseButton1Click:Connect(function()
    local Char = Player.Character or Player.CharacterAdded:Wait()
    local Hum = Char:WaitForChild("Humanoid")
    Hum.WalkSpeed = 250
    Hum.JumpPower = 120
    B3.Text = "✅ SUDAH KENCANG"
end)

-- ❌ TUTUP
B4.MouseButton1Click:Connect(function() SG:Destroy() end)

print("✅ RIKZ SCRIPT | ATTACK SPEED SIAP!")
